Store<AppState> createStore(SharedPreferences prefs) {
  return Store<AppState>(
    appReducer,
    middleware: [
      ...createLocalStorageMiddleware(prefs),
      ...createKyooAPIMiddleware(),
      ...createNavigationMiddleware()
    ],
    initialState: AppState.initState()
  );
}
